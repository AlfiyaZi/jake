 19 var fs = require('fs')
 20   , exec = require('child_process').exec
 21   , currDir = process.cwd();
 22 
 23 var NpmPublishTask = function (name, packageFiles) {
 24   this.name = name;
 25   this.packageFiles = packageFiles;
 26   this.define();
 27 };
 28 
 29 
 30 NpmPublishTask.prototype = new (function () {
 31 
 32   var getPackage = function () {
 33     var pkg = JSON.parse(fs.readFileSync(process.cwd() + '/package.json').toString());
 34     return pkg;
 35   };
 36 
 37   var getPackageVersionNumber = function () {
 38     return getPackage().version;
 39   };
 40 
 41   this.define = function () {
 42     var self = this;
 43 
 44     namespace('npm', function () {
 45      task('version', function () {
 46         // Only bump, push, and tag if the Git repo is clean
 47         exec('git status --porcelain --untracked-files=no',
 48             function (err, stdout, stderr) {
 49           var cmds
 50             , path
 51             , pkg
 52             , version
 53             , arr
 54             , patch
 55             , message;
 56 
 57           if (err) {
 58             fail(err);
 59           }
 60           if (stderr) {
 61             fail(new Error(stderr));
 62           }
 63           if (stdout.length) {
 64             fail(new Error('Git repository is not clean.'));
 65           }
 66 
 67           // Grab the current version-string
 68           path = process.cwd() + '/package.json';
 69           pkg = getPackage();
 70           version = pkg.version;
 71           // Increment the patch-number for the version
 72           arr = version.split('.');
 73           patch = parseInt(arr.pop(), 10) + 1;
 74           arr.push(patch);
 75           version = arr.join('.');
 76           // New package-version
 77           pkg.version = version;
 78           // Commit-message
 79           message = 'Version ' + version
 80 
 81           // Update package.json with the new version-info
 82           fs.writeFileSync(path, JSON.stringify(pkg, true, 2));
 83 
 84           cmds = [
 85             'git commit package.json -m "' + message + '"'
 86           , 'git push origin master'
 87           , 'git tag -a v' + version + ' -m "' + message + '"'
 88           , 'git push --tags'
 89           ];
 90 
 91           jake.exec(cmds, function () {
 92             var version = getPackageVersionNumber();
 93             console.log('Bumped version number to v' + version + '.');
 94             complete();
 95           });
 96         });
 97       }, {async: true});
 98 
 99       task('definePackage', function () {
100         var version = getPackageVersionNumber()
101           , t;
102         t = new jake.PackageTask(self.name, 'v' + version, function () {
103           this.packageFiles.include(self.packageFiles);
104           this.needTarGz = true;
105         });
106       });
107 
108       task('package', function () {
109         var definePack = jake.Task['npm:definePackage']
110           , pack = jake.Task['package']
111           , version = getPackageVersionNumber();
112         // May have already been run
113         definePack.reenable(true);
114         definePack.addListener('complete', function () {
115           pack.addListener('complete', function () {
116             console.log('Created package for ' + self.name + ' v' + version);
117             complete();
118           });
119           pack.invoke();
120         });
121         definePack.invoke();
122       }, {async: true});
123 
124       task('publish', function () {
125         var version = getPackageVersionNumber();
126         console.log('Publishing ' + self.name + ' v' + version);
127         cmds = [
128           'npm publish pkg/' + self.name + '-v' + version + '.tar.gz'
129         ];
130         // Hackity hack -- NPM publish sometimes returns errror like:
131         // Error sending version data\nnpm ERR!
132         // Error: forbidden 0.2.4 is modified, should match modified time
133         setTimeout(function () {
134           jake.exec(cmds, function () {
135             console.log('Published to NPM');
136             complete();
137           }, {stdout: true});
138         }, 5000);
139       }, {async: true});
140 
141       task('cleanup', function () {
142         var clobber = jake.Task['clobber'];
143         clobber.reenable(true);
144         clobber.invoke();
145         console.log('Cleaned up package');
146       });
147 
148     });
149 
150     desc('Bump version-number, package, and publish to NPM.');
151     task('publish', ['npm:version', 'npm:package',
152 153         'npm:publish', 'npm:cleanup'], function () {});
154   };
155 
156 })();
157 
158 jake.NpmPublishTask = NpmPublishTask;
159 exports.NpmPublishTask = NpmPublishTask;
