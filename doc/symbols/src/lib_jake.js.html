 19 var jake
 20   , fs = require('fs')
 21   , path = require('path')
 22   , task = require('./task')
 23   , Task = task.Task
 24   , FileTask = task.FileTask
 25   , DirectoryTask = task.DirectoryTask;
 26 
 27 var Namespace = function (name, parentNamespace) {
 28   this.name = name;
 29   this.parentNamespace = parentNamespace;
 30   this.childNamespaces = {};
 31   this.tasks = {};
 32 };
 33 
 34 var Invocation = function (taskName, args) {
 35   this.taskName = taskName;
 36   this.args = args;
 37 };
 38 
 39 /**
 40  * @namespace
 41  * The main namespace for Jake
 42  */
 43 jake = new function () {
 44 
 45   this._invocationChain = [];
 46 
 47   // Private variables
 48   // =================
 49   // Local reference for scopage
 50   var self = this;
 51 
 52   // Public properties
 53   // =================
 54   this.errorCode = undefined;
 55   // Name/value map of all the various tasks defined in a Jakefile.
 56   // Non-namespaced tasks are placed into 'default.'
 57   this.defaultNamespace = new Namespace('default', null);
 58   // For namespaced tasks -- tasks with no namespace are put into the
 59   // 'default' namespace so lookup code can work the same for both
 60   // namespaced and non-namespaced.
 61   this.currentNamespace = this.defaultNamespace;
 62   // Saves the description created by a 'desc' call that prefaces a
 63   // 'task' call that defines a task.
 64   this.currentTaskDescription = null;
 65 
 66   this.parseAllTasks = function () {
 67     var _parseNs = function (name, ns) {
 68       var nsTasks = ns.tasks
 69         , task
 70         , nsNamespaces = ns.childNamespaces
 71         , fullName;
 72       // Iterate through the tasks in each namespace
 73       for (var q in nsTasks) {
 74         task = nsTasks[q];
 75         // Preface only the namespaced tasks
 76         fullName = name == 'default' ? q : name + ':' + q;
 77         // Save with 'taskname' or 'namespace:taskname' key
 78         task.fullName = fullName;
 79         jake.Task[fullName] = task;
 80       }
 81       for (var p in nsNamespaces) {
 82         fullName = name  == 'default' ? p : name + ':' + p;
 83         _parseNs(fullName, nsNamespaces[p]);
 84       }
 85     };
 86 
 87     _parseNs('default', jake.defaultNamespace);
 88   };
 89 
 90   /**
 91    * Displays the list of descriptions avaliable for tasks defined in
 92    * a Jakefile
 93    */
 94   this.showAllTaskDescriptions = function (f) {
 95     var maxTaskNameLength = 0
 96       , task
 97       , str = ''
 98       , padding
 99       , name
100       , descr
101       , filter = typeof f == 'string' ? f : null;
102 
103     for (var p in jake.Task) {
104       task = jake.Task[p];
105       // Record the length of the longest task name -- used for
106       // pretty alignment of the task descriptions
107       maxTaskNameLength = p.length > maxTaskNameLength ?
108         p.length : maxTaskNameLength;
109     }
110     // Print out each entry with descriptions neatly aligned
111     for (var p in jake.Task) {
112       if (filter && p.indexOf(filter) == -1) {
113         continue;
114       }
115       task = jake.Task[p];
116 
117       name = '\033[32m' + p + '\033[39m ';
118 
119       // Create padding-string with calculated length
120       padding = (new Array(maxTaskNameLength - p.length + 2)).join(' ');
121 
122       descr = task.description
123       if (descr) {
124         descr = '\033[90m # ' + descr + '\033[39m \033[37m \033[39m';
125         console.log('jake ' + name + padding + descr);
126 127       }
128     }
129   };
130 
131   this.createTask = function () {
132     var args = Array.prototype.slice.call(arguments)
133       , task
134       , type
135       , name
136       , action
137       , opts
138       , prereqs = [];
139 
140       type = args.shift()
141 
142     // name, [deps], [action]
143     // Name (string) + deps (array) format
144     if (typeof args[0] == 'string') {
145       name = args.shift();
146       if (Array.isArray(args[0])) {
147         prereqs = args.shift();
148       }
149       if (typeof args[0] == 'function') {
150         action = args.shift();
151         opts =  args.shift() || {};
152       }
153     }
154     // name:deps, [action]
155     // Legacy object-literal syntax, e.g.: {'name': ['depA', 'depB']}
156     else {
157       obj = args.shift()
158       for (var p in obj) {
159         prereqs = prereqs.concat(obj[p]);
160         name = p;
161       }
162       action = args.shift();
163       opts =  args.shift() || {};
164     }
165 
166     switch (type) {
167       case 'directory':
168         action = function () {
169           if (!path.existsSync(name)) {
170             fs.mkdirSync(name, 0755);
171           }
172         };
173         task = new DirectoryTask(name, prereqs, action, opts);
174         break;
175       case 'file':
176         task = new FileTask(name, prereqs, action, opts);
177         break;
178       default:
179         task = new Task(name, prereqs, action, opts);
180     }
181 
182     if (jake.currentTaskDescription) {
183       task.description = jake.currentTaskDescription;
184       jake.currentTaskDescription = null;
185     }
186     jake.currentNamespace.tasks[name] = task;
187 
188     // FIXME: Should only need to add a new entry for the current
189     // task-definition, not reparse the entire structure
190     jake.parseAllTasks();
191   };
192 
193 }();

198 jake.Namespace = Namespace;
