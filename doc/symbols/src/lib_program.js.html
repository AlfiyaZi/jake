 19 var parseargs = require('../lib/parseargs')
 20   , Program
 21   , optsReg
 22   , preempts
 23   , usage
 24   , die;
 25 
 26 optsReg = [
 27   { full: 'jakefile'
 28   , abbr: 'f'
 29   , preempts: false
 30   , expectValue: true
 31   }
 32 , { full: 'directory'
 33   , abbr: 'C'
 34   , preempts: false
 35   , expectValue: true
 36   }
 37 , { full: 'always-make'
 38   , abbr: 'B'
 39   , preempts: false
 40   , expectValue: false
 41   }
 42 , { full: 'tasks'
 43   , abbr: 'T'
 44   , preempts: true
 45   }
 46 , { full: 'trace'
 47   , abbr: 't'
 48   , preempts: false
 49   , expectValue: false
 50   }
 51 , { full: 'help'
 52   , abbr: 'h'
 53   , preempts: true
 54   }
 55 , { full: 'version'
 56   , abbr: 'V'
 57   , preempts: true
 58   }
 59 ];
 60 
 61 preempts = {
 62   version: function () {
 63     die(jake.version);
 64   }
 65 , help: function () {
 66     die(usage);
 67   }
 68 };
 69 
 70 usage = ''
 71     + 'Jake JavaScript build tool\n'
 72     + '********************************************************************************\n'
 73     + 'If no flags are given, Jake looks for a Jakefile or Jakefile.js in the current directory.\n'
 74     + '********************************************************************************\n'
 75     + '{Usage}: jake [options ...] [env variables ...] target\n'
 76     + '\n'
 77     + '{Options}:\n'
 78     + '  -f, --jakefile FILE            Use FILE as the Jakefile\n'
 79     + '  -C, --directory DIRECTORY      Change to DIRECTORY before running tasks.\n'
 80     + '  -B, --always-make              Unconditionally make all targets.\n'
 81     + '  -T, --tasks                    Display the tasks, with descriptions, then exit.\n'
 82     + '  -t, --trace                    Enable full backtrace.\n'
 83     + '  -h, --help                     Outputs help information\n'
 84     + '  -V, --version                  Outputs Jake version\n'
 85     + '';
 86 
 87 Program = function () {
 88   this.opts = {};
 89   this.taskNames = null;
 90   this.taskArgs = null;
 91   this.envVars = null;
 92 };
 93 
 94 Program.prototype = new (function () {
 95   this.handleErr = function (err) {
 96     var msg;
 97     console.error('jake aborted.');
 98     if (this.opts.trace && err.stack) {
 99       console.error(err.stack);
100     }
101     else {
102       if (err.stack) {
103         msg = err.stack.split('\n').slice(0, 2).join('\n');
104         console.error(msg);
105         console.error('(See full trace by running task with --trace)');
106       }
107       else {
108         console.error(err.message);
109       }
110     }
111     process.exit(jake.errorCode || 1);
112   };
113 
114   this.parseArgs = function (args) {
115     var result = (new parseargs.Parser(optsReg)).parse(args);
116     this.opts = result.opts;
117     this.taskNames = result.taskNames;
118     this.envVars = result.envVars;
119   };
120 
121   this.preemptiveOption = function () {
122     var opts = this.opts;
123     for (var p in opts) {
124       if (preempts[p]) {
125         preempts[p]();
126         return true;
127       }
128     }
129     return false;
130   };
131 
132 })();
133 
134 die = function (msg) {
135   console.log(msg);
136   process.exit();
137 };
138 
139 module.exports.Program = Program;
