 20 var exec = require('child_process').exec
 21   , spawn = require('child_process').spawn
 22   , utils;
 23 
 24 utils = new (function () {
 25   var _mix = function (targ, src, merge, includeProto) {
 26         for (var p in src) {
 27           // Don't copy stuff from the prototype
 28           if (src.hasOwnProperty(p) || includeProto) {
 29             if (merge &&
 30                 // Assumes the source property is an Object you can
 31                 // actually recurse down into
 32                 (typeof src[p] == 'object') &&
 33                 (src[p] !== null) &&
 34                 !(src[p] instanceof Array)) {
 35               // Create the source property if it doesn't exist
 36               // TODO: What if it's something weird like a String or Number?
 37               if (typeof targ[p] == 'undefined') {
 38                 targ[p] = {};
 39               }
 40               _mix(targ[p], src[p], merge, includeProto); // Recurse
 41             }
 42             // If it's not a merge-copy, just set and forget
 43             else {
 44               targ[p] = src[p];
 45             }
 46           }
 47         }
 48       }
 49 
 50     , _trim = function (s) {
 51         var str = s || '';
 52         return str.replace(/^\s*|\s*$/g, '');
 53       }
 54 
 55     , _truncate = function (s) {
 56         var str = s ? s.replace(/\n$/, '') : '';
 57         return str;
 58       };
 59 
 60 
 61   this.exec = function (arr, callback, opts) {
 62     var options = opts || {}
 63       , list = arr.slice()
 64       , stdout = options.stdout
 65       , stderr = options.stderr
 66       , breakOnError = typeof options.breakOnError != 'undefined' ?
 67             options.breakOnError : true
 68       , run;
 69 
 70     run = function () {
 71       var sh
 72         , cmd
 73         , args
 74         , next = list.shift()
 75         , errData = '';
 76       // Keep running as long as there are commands in the array
 77       if (next) {
 78         // Ganking part of Node's child_process.exec to get cmdline args parsed
 79         if (process.platform == 'win32') {
 80           cmd = 'cmd.exe';
 81           // TODO: Escape double-quotes?
 82           args = ['/s', '/c', '"' + next + '"'];
 83         }
 84         else {
 85           cmd = '/bin/sh';
 86           args = ['-c', next];
 87         }
 88 
 89         // Spawn a child-process, set up output
 90         sh = spawn(cmd, args);
 91         // Out
 92         if (stdout) {
 93           sh.stdout.on('data', function (data) {
 94             console.log(_truncate(data.toString()));
 95           });
 96         }
 97         // Err
 98         sh.stderr.on('data', function (data) {
 99           var d = data.toString();
100           if (stderr) {
101             console.error(_truncate(d));
102           }
103           // Accumulate the error-data so we can use it as the
104           // stack if the process exits with an error
105           errData += d;
106         });
107         // Exit, handle err or run next
108         sh.on('exit', function (code) {
109           var msg = errData || 'Process exited with error.';
110           msg = _trim(msg);
111           if (breakOnError && code != 0) {
112             fail(msg, code);
113           }
114           else {
115             run();
116           }
117         });
118       }
119       else {
120         if (callback) {
121           callback();
122         }
123       }
124     };
125 
126     run();
127   };
128 
129   this.objectToString = function (object) {
130     var objectArray = [];
131     for (var key in object) {
132       if ('object' == typeof object[key]) {
133         objectArray.push(this.objectToString(object[key]));
134       } else {
135         objectArray.push(key + '=' + object[key]);
136       }
137     }
138     return objectArray.join(', ');
139   };
140 
141   /*
142    * Mix in the properties on an object to another object
143    * yam.mixin(target, source, [source,] [source, etc.] [merge-flag]);
144    * 'merge' recurses, to merge object sub-properties together instead
145    * of just overwriting with the source object.
146    */
147   this.mixin = (function () {
148     return function () {
149       var args = Array.prototype.slice.apply(arguments),
150           merge = false,
151           targ, sources;
152       if (args.length > 2) {
153         if (typeof args[args.length - 1] == 'boolean') {
154           merge = args.pop();
155         }
156       }
157       targ = args.shift();
158       sources = args;
159       for (var i = 0, ii = sources.length; i < ii; i++) {
160         _mix(targ, sources[i], merge);
161       }
162       return targ;
163     };
164   }).call(this);
165 
166   this.enhance = (function () {
167     return function () {
168       var args = Array.prototype.slice.apply(arguments),
169           merge = false,
170           targ, sources;
171       if (args.length > 2) {
172         if (typeof args[args.length - 1] == 'boolean') {
173           merge = args.pop();
174         }
175       }
176       targ = args.shift();
177       sources = args;
178       for (var i = 0, ii = sources.length; i < ii; i++) {
179         _mix(targ, sources[i], merge, true);
180       }
181       return targ;
182     };
183   }).call(this);
184 
185 })();
186 
187 module.exports = utils;
