 19 var path = require('path')
 20   , fs = require('fs')
 21   , exec = require('child_process').exec
 22   , currDir = process.cwd();
 23 
 24 var PackageTask = function (name, version, definition) {
 25   this.name = name;
 26   this.version = version;
 27   this.packageDir = 'pkg';
 28   this.packageFiles = new jake.FileList();
 29   this.tag = false;
 30   this.needTar = false;
 31   this.needTarGz = false;
 32   this.needTarBz2 = false;
 33   this.tarCommand = 'tar';
 34   this.zipCommand = 'zip';
 35   if (typeof definition == 'function') {
 36     definition.call(this);
 37   }
 38   this.define();
 39 };
 40 
 41 PackageTask.prototype = new (function () {
 42 
 43   var _compressOpts = {
 44         Tar: {
 45           ext: '.tgz'
 46         , flag: 'z'
 47         }
 48       , TarGz: {
 49           ext: '.tar.gz'
 50         , flag: 'z'
 51         }
 52       , TarBz2: {
 53           ext: '.tar.bz2'
 54         , flag: 'j'
 55         }
 56       };
 57 
 58   this.define = function () {
 59     var self = this
 60       , packageDirPath = this.packageDirPath()
 61       , compressTaskArr = [];
 62 
 63     desc('Build the package for distribution');
 64     task('package', ['clobberPackage', 'buildPackage']);
 65     // Backward-compat alias
 66     task('repackage', ['package']);
 67 
 68     task('clobberPackage', function () {
 69       jake.exec(['rm -fr ' + self.packageDir], function () {
 70         complete();
 71       });
 72     }, {async: true});
 73 
 74     desc('Remove the package');
 75     task('clobber', ['clobberPackage']);
 76 
 77     for (var p in _compressOpts) {
 78       if (this['need' + p]) {
 79         (function (p) {
 80           var filename = self.packageDir + '/' + self.packageName() +
 81               _compressOpts[p].ext;
 82           compressTaskArr.push(filename);
 83 
 84           file(filename, [packageDirPath], function () {
 85             var opts = _compressOpts[p];
 86             // Move into the package dir to compress
 87             process.chdir(self.packageDir);
 88             var cmd = self.tarCommand + ' -' + opts.flag + 'cvf ' +
 89                 self.packageName() + opts.ext + ' ' + self.packageName();
 90             exec(cmd, function (err, stdout, stderr) {
 91               if (err) { throw err; }
 92               // Return back up to the project directory
 93               process.chdir(currDir);
 94               complete();
 95             });
 96           }, {async: true});
 97         })(p);
 98       }
 99     }
100 
101     task('buildPackage', compressTaskArr, function () {});
102 
103     directory(this.packageDir);
104 
105     file(packageDirPath,
106 107           [this.packageDir].concat(self.packageFiles.toArray()), function () {
108       var fileList = [];
109       self.packageFiles.forEach(function (name) {
110         var f = path.join(self.packageDirPath(), name)
111           , fDir = path.dirname(f)
112           , fDirArr = fDir.split('/')
113           , baseDir = ''
114           , stats;
115 
116         // Make any necessary container directories
117         fDirArr.forEach(function (dir) {
118           baseDir += baseDir ? '/' + dir : dir;
119           if (!path.existsSync(baseDir)) {
120             fs.mkdirSync(baseDir, 0755);
121 122           }
123         });
124 
125         // Add both files and directories, will be copied with -R
126         fileList.push({
127           to: name
128         , from: f
129         });
130       });
131       var _copyFile = function () {
132         var cmd
133           , file = fileList.pop();
134         if (file) {
135           // Do recursive copy of files and directories
136           cmd = 'cp -R ' + file.to + ' ' + file.from;
137           exec(cmd, function (err, stdout, stderr) {
138             if (err) { throw err; }
139             _copyFile();
140           });
141         }
142         else {
143           complete();
144         }
145       };
146       _copyFile();
147     }, {async: true});
148 
149 
150   };
151 
152   this.packageName = function () {
153     if (this.version) {
154       return this.name + '-' + this.version;
155     }
156     else {
157       return this.name;
158     }
159   };
160 
161   this.packageDirPath = function () {
162     return this.packageDir + '/' + this.packageName();
163   };
164 
165 })();
166 
167 jake.PackageTask = PackageTask;
168 exports.PackageTask = PackageTask;
