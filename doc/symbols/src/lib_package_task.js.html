 19 var path = require('path')
 20   , fs = require('fs')
 21   , exec = require('child_process').exec
 22   , currDir = process.cwd();
 23 
 24 /**
 25   @name jake
 26   @namespace jake
 27 */
 28 /**
 29   @name jake.PackageTask
 30   @constructor
 31   @description Instantiating a PackageTask creates a number of Jake
 32   Tasks that make packaging and distributing your software easy.
 33 
 34   @param {String} name The name of the project
 35   @param {String} version The current project version (will be
 36   appended to the project-name in the package-archive
 37   @param {Function} definition Defines the contents of the package,
 38   and format of the package-archive. Will be executed on the instantiated
 39   PackageTask (i.e., 'this', will be the PackageTask instance),
 40   to set the various instance-propertiess.
 41 
 42   @example
 43   var t = new jake.PackageTask('rous', 'v' + version, function () {
 44     var files = [
 45       'Capfile'
 46     , 'Jakefile'
 47     , 'README.md'
 48     , 'package.json'
 49     , 'app/*'
 50     , 'bin/*'
 51     , 'config/*'
 52     , 'lib/*'
 53     , 'node_modules/*'
 54     ];
 55     this.packageFiles.include(files);
 56     this.packageFiles.exclude('node_modules/foobar');
 57     this.needTarGz = true;
 58   });
 59 
 60  */
 61 var PackageTask = function (name, version, definition) {
 62   /**
 63     @name jake.PackageTask#name
 64     @public
 65     @type {String}
 66     @description The name of the project
 67    */
 68   this.name = name;
 69   /**
 70     @name jake.PackageTask#version
 71     @public
 72     @type {String}
 73     @description The project version-string
 74    */
 75   this.version = version;
 76   /**
 77     @name jake.PackageTask#version
 78     @public
 79     @type {String='pkg'}
 80     @description The directory-name to use for packaging the software
 81    */
 82   this.packageDir = 'pkg';
 83   /**
 84     @name jake.PackageTask#packageFiles
 85     @public
 86     @type {jake.FileList}
 87     @description The list of files and directories to include in the
 88     package-archive
 89    */
 90   this.packageFiles = new jake.FileList();
 91   /**
 92     @name jake.PackageTask#needTar
 93     @public
 94     @type {Boolean=false}
 95     @description If set to true, uses the `tar` utility to create
 96     a gzip .tgz archive of the pagckage
 97    */
 98   this.needTar = false;
 99   /**
100     @name jake.PackageTask#needTar
101     @public
102     @type {Boolean=false}
103     @description If set to true, uses the `tar` utility to create
104     a gzip .tar.gz archive of the pagckage
105    */
106   this.needTarGz = false;
107   /**
108     @name jake.PackageTask#needTarBz2
109     @public
110     @type {Boolean=false}
111     @description If set to true, uses the `tar` utility to create
112     a bzip2 .bz2 archive of the pagckage
113    */
114   this.needTarBz2 = false;
115   /**
116     @name jake.PackageTask#needJar
117     @public
118     @type {Boolean=false}
119     @description If set to true, uses the `jar` utility to create
120     a .jar archive of the pagckage
121    */
122   this.needJar = false;
123   /**
124     @name jake.PackageTask#needZip
125     @public
126     @type {Boolean=false}
127     @description If set to true, uses the `zip` utility to create
128     a .zip archive of the pagckage
129    */
130   this.needZip = false;
131   /**
132     @name jake.PackageTask#manifestFile
133     @public
134     @type {String=null}
135     @description Can be set to point the `jar` utility at a manifest
136     file to use in a .jar archive. If unset, one will be automatically
137     created by the `jar` utility. This path should be relative to the
138     root of the package directory (this.packageDir above, likely 'pkg')
139    */
140   this.manifestFile = null;
141   /**
142     @name jake.PackageTask#tarCommand
143     @public
144     @type {String='tar'}
145     @description The shell-command to use for creating tar archives.
146    */
147   this.tarCommand = 'tar';
148   /**
149     @name jake.PackageTask#jarCommand
150     @public
151     @type {String='jar'}
152     @description The shell-command to use for creating jar archives.
153    */
154   this.jarCommand = 'jar';
155   /**
156     @name jake.PackageTask#zipCommand
157     @public
158     @type {String='zip'}
159     @description The shell-command to use for creating zip archives.
160    */
161   this.zipCommand = 'zip';
162   /**
163     @name jake.PackageTask#archiveChangeDir
164     @public
165     @type {String=null}
166     @description Equivalent to the '-C' command for the `tar` and `jar`
167     commands. ("Change to this directory before adding files.")
168    */
169   this.archiveChangeDir = null;
170   /**
171     @name jake.PackageTask#archiveContentDir
172     @public
173     @type {String=null}
174     @description Specifies the files and directories to include in the
175     package-archive. If unset, this will default to the main package
176     directory -- i.e., name + version.
177    */
178   this.archiveContentDir = null;
179 
180   if (typeof definition == 'function') {
181     definition.call(this);
182   }
183   this.define();
184 };
185 
186 PackageTask.prototype = new (function () {
187 
188   var _compressOpts = {
189         Tar: {
190           ext: '.tgz'
191         , flags: 'cvzf'
192         , cmd: 'tar'
193         }
194       , TarGz: {
195           ext: '.tar.gz'
196         , flags: 'cvzf'
197         , cmd: 'tar'
198         }
199       , TarBz2: {
200           ext: '.tar.bz2'
201         , flags: 'cvjf'
202         , cmd: 'tar'
203         }
204       , Jar: {
205           ext: '.jar'
206         , flags: 'cf'
207         , cmd: 'jar'
208         }
209       , Zip: {
210           ext: '.zip'
211         , flags: 'r'
212         , cmd: 'zip'
213         }
214       };
215 
216   this.define = function () {
217     var self = this
218       , packageDirPath = this.packageDirPath()
219       , compressTaskArr = [];
220 
221     desc('Build the package for distribution');
222     task('package', ['clobberPackage', 'buildPackage']);
223     // Backward-compat alias
224     task('repackage', ['package']);
225 
226     task('clobberPackage', function () {
227       jake.exec(['rm -fr ' + self.packageDir], function () {
228         complete();
229       });
230     }, {async: true});
231 
232     desc('Remove the package');
233     task('clobber', ['clobberPackage']);
234 
235     for (var p in _compressOpts) {
236       if (this['need' + p]) {
237         (function (p) {
238           var filename = self.packageDir + '/' + self.packageName() +
239               _compressOpts[p].ext;
240           compressTaskArr.push(filename);
241 
242           file(filename, [packageDirPath], function () {
243             var cmd
244               , opts = _compressOpts[p];
245 
246             // Move into the package dir to compress (see below, after
247             // exec)
248             process.chdir(self.packageDir);
249 
250             cmd = self[opts.cmd + 'Command'];
251             cmd += ' -' + opts.flags;
252             if (opts.cmd == 'jar' && self.manifestFile) {
253               cmd += 'm';
254             }
255             cmd += ' ' + self.packageName() + opts.ext;
256             if (opts.cmd == 'jar' && self.manifestFile) {
257               cmd += ' ' + self.manifestFile;
258             }
259             if (self.archiveChangeDir) {
260               cmd += ' -C ' + self.archiveChangeDir;
261             }
262             if (self.archiveContentDir) {
263               cmd += ' ' + self.archiveContentDir;
264             }
265             else {
266               cmd += ' ' + self.packageName();
267             }
268 
269             exec(cmd, function (err, stdout, stderr) {
270               if (err) { throw err; }
271 
272               // Return back up to the project directory (see above,
273               // before exec)
274               process.chdir(currDir);
275 
276               complete();
277             });
278           }, {async: true});
279         })(p);
280       }
281     }
282 
283     task('buildPackage', compressTaskArr, function () {});
284 
285     directory(this.packageDir);
286 
287     file(packageDirPath,
288           [this.packageDir].concat(self.packageFiles.toArray()), function () {
289       var fileList = [];
290       self.packageFiles.forEach(function (name) {
291         var f = path.join(self.packageDirPath(), name)
292           , fDir = path.dirname(f)
293           , fDirArr = fDir.split('/')
294           , baseDir = ''
295           , stats;
296 
297         // Make any necessary container directories
298         fDirArr.forEach(function (dir) {
299           baseDir += baseDir ? '/' + dir : dir;
300           if (!path.existsSync(baseDir)) {
301             fs.mkdirSync(baseDir, 0755);
302           }
303         });
304 
305         // Add both files and directories, will be copied with -R
306         fileList.push({
307           to: name
308         , from: f
309         });
310       });
311       var _copyFile = function () {
312         var cmd
313           , file = fileList.pop();
314         if (file) {
315           // Do recursive copy of files and directories
316           cmd = 'cp -R ' + file.to + ' ' + file.from;
317           exec(cmd, function (err, stdout, stderr) {
318             if (err) { throw err; }
319             _copyFile();
320           });
321         }
322         else {
323           complete();
324         }
325       };
326       _copyFile();
327     }, {async: true});
328 
329 
330   };
331 
332   this.packageName = function () {
333     if (this.version) {
334       return this.name + '-' + this.version;
335     }
336     else {
337       return this.name;
338     }
339   };
340 
341   this.packageDirPath = function () {
342     return this.packageDir + '/' + this.packageName();
343   };
344 
345 })();
346 
347 jake.PackageTask = PackageTask;
348 exports.PackageTask = PackageTask;
