 20 var exec = require('child_process').exec
 21   , spawn = require('child_process').spawn
 22   , fileUtils = require('./file')
 23   , utils;
 24 
 25 /**
 26   @name jake
 27   @namespace jake
 28 */
 29 utils = new (function () {
 30   var _mix = function (targ, src, merge, includeProto) {
 31         for (var p in src) {
 32           // Don't copy stuff from the prototype
 33           if (src.hasOwnProperty(p) || includeProto) {
 34             if (merge &&
 35                 // Assumes the source property is an Object you can
 36                 // actually recurse down into
 37                 (typeof src[p] == 'object') &&
 38                 (src[p] !== null) &&
 39                 !(src[p] instanceof Array)) {
 40               // Create the source property if it doesn't exist
 41               // TODO: What if it's something weird like a String or Number?
 42               if (typeof targ[p] == 'undefined') {
 43                 targ[p] = {};
 44               }
 45               _mix(targ[p], src[p], merge, includeProto); // Recurse
 46             }
 47             // If it's not a merge-copy, just set and forget
 48             else {
 49               targ[p] = src[p];
 50             }
 51           }
 52         }
 53       }
 54 
 55     , _trim = function (s) {
 56         var str = s || '';
 57         return str.replace(/^\s*|\s*$/g, '');
 58       }
 59 
 60     , _truncate = function (s) {
 61         var str = s ? s.replace(/\n$/, '') : '';
 62         return str;
 63       };
 64 
 65 
 66   /**
 67     @name jake.exec
 68     @static
 69     @function
 70     @description Executes shell-commands asynchronously with an optional
 71     final callback.
 72     `
 73     @param {String[]} cmds The list of shell-commands to execute
 74     @param {Function} [callback] Callback to run after executing  the
 75     commands
 76     @param {Object} [opts]
 77       @param {Boolean} [opts.stdout=false] Print stdout from each command
 78       @param {Boolean} [opts.stderr=false] Print stderr from each command
 79       @param {Boolean} [opts.breakOnError=true] Stop further execution on
 80       the first error.
 81 
 82     @example
 83     var cmds = [
 84           'echo "showing directories"'
 85         , 'ls -al | grep ^d'
 86         , 'echo "moving up a directory"'
 87         , 'cd ../'
 88         ]
 89       , callback = function () {
 90           console.log('Finished running commands.');
 91         }
 92     jake.exec(cmds, callback, {stdout: true});
 93    */
 94   this.exec = function () {
 95     var args
 96       , arg
 97       , list
 98       , callback
 99       , opts = {}
100       , printStdout = false
101       , printStderr = false
102       , breakOnError = true
103       , run;
104 
105     args = Array.prototype.slice.call(arguments);
106 
107     // The list of shell-commands to run (make a copy)
108     list = args.shift().slice();
109 
110     // Get optional callback or opts
111     while((arg = args.shift())) {
112       if (typeof arg == 'function') {
113         callback = arg;
114       }
115       else if (typeof arg == 'object') {
116         opts = arg;
117       }
118     }
119 
120     // Override default options
121     printStdout = !!opts.stdout;
122     printStderr = !!opts.stderr;
123     breakOnError = typeof opts.breakOnError != 'undefined' ?
124           opts.breakOnError : true;
125 
126     run = function () {
127       var sh
128         , cmd
129         , args
130         , next = list.shift()
131         , errData = '';
132       // Keep running as long as there are commands in the array
133       if (next) {
134         // If you're on Windows, no streaming output for you,
135         // just use exec
136         if (process.platform == 'win32') {
137           exec(next, function (err, stdout, stderr) {
138             if (err && breakOnError) {
139               fail(err);
140             }
141             else {
142               if (printStderr) {
143                 console.error(stderr);
144               }
145               if (printStdout) {
146                 console.log(stdout);
147               }
148               run();
149             }
150           });
151         }
152         // POSIX platform gets streaming output for shell-commands
153         // Ganking part of Node's child_process.exec to get cmdline args parsed
154         else {
155           cmd = '/bin/sh';
156           args = ['-c', next];
157 
158           // Spawn a child-process, set up output
159           sh = spawn(cmd, args);
160           // Out
161           if (printStdout) {
162             sh.stdout.on('data', function (data) {
163               console.log(_truncate(data.toString()));
164             });
165           }
166           // Err
167           sh.stderr.on('data', function (data) {
168             var d = data.toString();
169             if (printStderr) {
170               console.error(_truncate(d));
171             }
172             // Accumulate the error-data so we can use it as the
173             // stack if the process exits with an error
174             errData += d;
175           });
176           // Exit, handle err or run next
177           sh.on('exit', function (code) {
178             var msg = errData || 'Process exited with error.';
179             msg = _trim(msg);
180             if (breakOnError && code != 0) {
181               fail(msg, code);
182             }
183             else {
184               run();
185             }
186           });
187         }
188       }
189       else {
190         if (typeof callback == 'function') {
191           callback();
192         }
193       }
194     };
195 
196     run();
197   };
198 
199   this.objectToString = function (object) {
200     var objectArray = [];
201     for (var key in object) {
202       if ('object' == typeof object[key]) {
203         objectArray.push(this.objectToString(object[key]));
204       } else {
205         objectArray.push(key + '=' + object[key]);
206       }
207     }
208     return objectArray.join(', ');
209   };
210 
211   /*
212    * Mix in the properties on an object to another object
213    * yam.mixin(target, source, [source,] [source, etc.] [merge-flag]);
214    * 'merge' recurses, to merge object sub-properties together instead
215    * of just overwriting with the source object.
216    */
217   this.mixin = (function () {
218     return function () {
219       var args = Array.prototype.slice.apply(arguments),
220           merge = false,
221           targ, sources;
222       if (args.length > 2) {
223         if (typeof args[args.length - 1] == 'boolean') {
224           merge = args.pop();
225         }
226       }
227       targ = args.shift();
228       sources = args;
229       for (var i = 0, ii = sources.length; i < ii; i++) {
230         _mix(targ, sources[i], merge);
231       }
232       return targ;
233     };
234   }).call(this);
235 
236   this.enhance = (function () {
237     return function () {
238       var args = Array.prototype.slice.apply(arguments),
239           merge = false,
240           targ, sources;
241       if (args.length > 2) {
242         if (typeof args[args.length - 1] == 'boolean') {
243           merge = args.pop();
244         }
245       }
246       targ = args.shift();
247       sources = args;
248       for (var i = 0, ii = sources.length; i < ii; i++) {
249         _mix(targ, sources[i], merge, true);
250       }
251       return targ;
252     };
253   }).call(this);
254 
255 })();

258 utils.mixin(utils, fileUtils);
