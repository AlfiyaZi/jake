 18 var exec = require('child_process').exec;

 20 /**
 21   @name jake
 22   @namespace jake
 23 */
 24 var api = new (function () {
 25   /**
 26     @name task
 27     @static
 28     @function
 29     @description Creates a Jake Task
 30     `
 31     @param {String} name The name of the Task
 32     @param {Array} [prereqs] Prerequisites to be run before this task
 33     @param {Function} [action] The action to perform for this task
 34     @param {Object} [opts]
 35       @param {Array} [opts.asyc=false] Perform this task asynchronously.
 36       If you flag a task with this option, you must call the global
 37       `complete` method inside the task's action, for execution to proceed
 38       to the next task.
 39 
 40     @example
 41     desc('This is the default task.');
 42     task('default', function (params) {
 43       console.log('This is the default task.');
 44     });
 45 
 46     desc('This task has prerequisites.');
 47     task('hasPrereqs', ['foo', 'bar', 'baz'], function (params) {
 48       console.log('Ran some prereqs first.');
 49     });
 50 
 51     desc('This is an asynchronous task.');
 52     task('asyncTask', function () {
 53       setTimeout(complete, 1000);
 54     }, {async: true});
 55    */
 56   this.task = function (name, prereqs, action, opts) {
 57     var args = Array.prototype.slice.call(arguments)
 58       , type;
 59     args.unshift('task');
 60     jake.createTask.apply(global, args);
 61     jake.currentTaskDescription = null;
 62   };
 63 
 64   /**
 65     @name directory
 66     @static
 67     @function
 68     @description Creates a Jake DirectoryTask. Can be used as a prerequisite
 69     for FileTasks, or for simply ensuring a directory exists for use with a
 70     Task's action.
 71     `
 72     @param {String} name The name of the DiretoryTask
 73 
 74     @example
 75 
 76     // Creates the package directory for distribution
 77     directory('pkg');
 78    */
 79   this.directory = function (name) {
 80     var args = Array.prototype.slice.call(arguments);
 81     args.unshift('directory');
 82     jake.createTask.apply(global, args);
 83     jake.currentTaskDescription = null;
 84   };
 85 
 86   /**
 87     @name file
 88     @static
 89     @function
 90     @description Creates a Jake FileTask.
 91     `
 92     @param {String} name The name of the FileTask
 93     @param {Array} [prereqs] Prerequisites to be run before this task
 94     @param {Function} [action] The action to create this file, if it doesn't
 95     exist already.
 96     @param {Object} [opts]
 97       @param {Array} [opts.asyc=false] Perform this task asynchronously.
 98       If you flag a task with this option, you must call the global
 99       `complete` method inside the task's action, for execution to proceed
100       to the next task.
101 
102    */
103   this.file = function (name, prereqs, action, opts) {
104     var args = Array.prototype.slice.call(arguments);
105     args.unshift('file');
106     jake.createTask.apply(global, args);
107     jake.currentTaskDescription = null;
108   };
109 
110   /**
111     @name desc
112     @static
113     @function
114     @description Creates a description for a Jake Task (or FileTask,
115     DirectoryTask). When invoked, the description that iscreated will
116     be associated with whatever Task is created next.
117     `
118     @param {String} description The description for the Task
119    */
120   this.desc = function (description) {
121     jake.currentTaskDescription = description;
122   };
123 
124   /**
125     @name namespace
126     @static
127     @function
128     @description Creates a namespace which allows logical grouping
129     of tasks, and prevents name-collisions with task-names. Namespaces
130     can be nested inside of other namespaces.
131     `
132     @param {String} name The name of the namespace
133     @param {Function} scope The enclosing scope for the namespaced tasks
134 
135     @example
136     namespace('doc', function () {
137       task('generate', ['doc:clobber'], function () {
138         // Generate some docs
139       });
140 
141       task('clobber', function () {
142         // Clobber the doc directory first
143       });
144     });
145    */
146   this.namespace = function (name, scope) {
147     var curr = jake.currentNamespace
148       , ns = new jake.Namespace(name, curr);
149     curr.childNamespaces[name] = ns;
150     jake.currentNamespace = ns;
151     scope();
152     jake.currentNamespace = curr;
153     jake.currentTaskDescription = null;
154   };
155 
156   /**
157     @name complete
158     @static
159     @function
160     @description Complets an asynchronous task, allowing Jake's
161     execution to proceed to the next task
162     `
163     @example
164     task('generate', ['doc:clobber'], function () {
165       exec('./generate_docs.sh', function (err, stdout, stderr) {
166         if (err || stderr) {
167           fail(err || stderr);
168         }
169         else {
170           console.log(stdout);
171           complete();
172         }
173       });
174     }, {async: true});
175    */
176   this.complete = function () {
177     var current = jake._invocationChain.pop();
178     if (current) {
179       current.complete();
180     }
181   };
182 
183   /**
184     @name fail
185     @static
186     @function
187     @description Causes Jake execution to abort with an error.
188     Allows passing an optional error code, which will be used to
189     set the exit-code of exiting process.
190     `
191     @param {Error|String} err The error to thow when aborting execution.
192     If this argument is an Error object, it will simply be thrown. If
193     a String, it will be used as the error-message. (If it is a multi-line
194     String, the first line will be used as the Error message, and the
195     remaining lines will be used as the error-stack.)
196 
197     @example
198     task('createTests, function () {
199       if (!path.existsSync('./tests')) {
200         fail('Test directory does not exist.');
201       }
202       else {
203         // Do some testing stuff ...
204       }
205     });
206    */
207   this.fail = function (err, code) {
208     var msg
209       , errObj;
210     if (code) {
211       jake.errorCode = code;
212     }
213     if (err) {
214       if (typeof err == 'string') {
215         // Use the initial or only line of the error as the error-message
216         // If there was a multi-line error, use the rest as the stack
217         msg = err.split('/n');
218         errObj = new Error(msg.shift());
219         if (msg.length) {
220           errObj.stack = msg.join('\n');
221         }
222         throw errObj;
223       }
224       else if (err instanceof Error) {
225         throw err;
226       }
227       else {
228         throw new Error(err.toString());
229       }
230     }
231     else {
232       throw new Error();
233     }
234   };
235 
236 })();
237
