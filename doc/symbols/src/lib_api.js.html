 18 var exec = require('child_process').exec;
 19 
 20 var api = new (function () {
 21   /**
 22     @name task
 23     @static
 24     @function
 25     @description Creates a Jake Task
 26     `
 27     @param {String} name The name of the Task
 28     @param {Array} [prereqs] Prerequisites to be run before this task
 29     @param {Function} [action] The action to perform for this task
 30     @param {Object} [opts]
 31       @param {Boolean} [opts.asyc=false] Perform this task asynchronously.
 32       If you flag a task with this option, you must call the global
 33       `complete` method inside the task's action, for execution to proceed
 34       to the next task.
 35 
 36     @example
 37     desc('This is the default task.');
 38     task('default', function (params) {
 39       console.log('This is the default task.');
 40     });
 41 
 42     desc('This task has prerequisites.');
 43     task('hasPrereqs', ['foo', 'bar', 'baz'], function (params) {
 44       console.log('Ran some prereqs first.');
 45     });
 46 
 47     desc('This is an asynchronous task.');
 48     task('asyncTask', function () {
 49       setTimeout(complete, 1000);
 50     }, {async: true});
 51    */
 52   this.task = function (name, prereqs, action, opts) {
 53     var args = Array.prototype.slice.call(arguments)
 54       , type;
 55     args.unshift('task');
 56     jake.createTask.apply(global, args);
 57     jake.currentTaskDescription = null;
 58   };
 59 
 60   /**
 61     @name directory
 62     @static
 63     @function
 64     @description Creates a Jake DirectoryTask. Can be used as a prerequisite
 65     for FileTasks, or for simply ensuring a directory exists for use with a
 66     Task's action.
 67     `
 68     @param {String} name The name of the DiretoryTask
 69 
 70     @example
 71 
 72     // Creates the package directory for distribution
 73     directory('pkg');
 74    */
 75   this.directory = function (name) {
 76     var args = Array.prototype.slice.call(arguments);
 77     args.unshift('directory');
 78     jake.createTask.apply(global, args);
 79     jake.currentTaskDescription = null;
 80   };
 81 
 82   /**
 83     @name file
 84     @static
 85     @function
 86     @description Creates a Jake FileTask.
 87     `
 88     @param {String} name The name of the FileTask
 89     @param {Array} [prereqs] Prerequisites to be run before this task
 90     @param {Function} [action] The action to create this file, if it doesn't
 91     exist already.
 92     @param {Object} [opts]
 93       @param {Array} [opts.asyc=false] Perform this task asynchronously.
 94       If you flag a task with this option, you must call the global
 95       `complete` method inside the task's action, for execution to proceed
 96       to the next task.
 97 
 98    */
 99   this.file = function (name, prereqs, action, opts) {
100     var args = Array.prototype.slice.call(arguments);
101     args.unshift('file');
102     jake.createTask.apply(global, args);
103     jake.currentTaskDescription = null;
104   };
105 
106   /**
107     @name desc
108     @static
109     @function
110     @description Creates a description for a Jake Task (or FileTask,
111     DirectoryTask). When invoked, the description that iscreated will
112     be associated with whatever Task is created next.
113     `
114     @param {String} description The description for the Task
115    */
116   this.desc = function (description) {
117     jake.currentTaskDescription = description;
118   };
119 
120   /**
121     @name namespace
122     @static
123     @function
124     @description Creates a namespace which allows logical grouping
125     of tasks, and prevents name-collisions with task-names. Namespaces
126     can be nested inside of other namespaces.
127     `
128     @param {String} name The name of the namespace
129     @param {Function} scope The enclosing scope for the namespaced tasks
130 
131     @example
132     namespace('doc', function () {
133       task('generate', ['doc:clobber'], function () {
134         // Generate some docs
135       });
136 
137       task('clobber', function () {
138         // Clobber the doc directory first
139       });
140     });
141    */
142   this.namespace = function (name, scope) {
143     var curr = jake.currentNamespace
144       , ns = new jake.Namespace(name, curr);
145     curr.childNamespaces[name] = ns;
146     jake.currentNamespace = ns;
147     scope();
148     jake.currentNamespace = curr;
149     jake.currentTaskDescription = null;
150   };
151 
152   /**
153     @name complete
154     @static
155     @function
156     @description Complets an asynchronous task, allowing Jake's
157     execution to proceed to the next task
158     `
159     @example
160     task('generate', ['doc:clobber'], function () {
161       exec('./generate_docs.sh', function (err, stdout, stderr) {
162         if (err || stderr) {
163           fail(err || stderr);
164         }
165         else {
166           console.log(stdout);
167           complete();
168         }
169       });
170     }, {async: true});
171    */
172   this.complete = function () {
173     var current = jake._invocationChain.pop();
174     if (current) {
175       current.complete();
176     }
177   };
178 
179   /**
180     @name fail
181     @static
182     @function
183     @description Causes Jake execution to abort with an error.
184     Allows passing an optional error code, which will be used to
185     set the exit-code of exiting process.
186     `
187     @param {Error|String} err The error to thow when aborting execution.
188     If this argument is an Error object, it will simply be thrown. If
189     a String, it will be used as the error-message. (If it is a multi-line
190     String, the first line will be used as the Error message, and the
191     remaining lines will be used as the error-stack.)
192 
193     @example
194     task('createTests, function () {
195       if (!path.existsSync('./tests')) {
196         fail('Test directory does not exist.');
197       }
198       else {
199         // Do some testing stuff ...
200       }
201     });
202    */
203   this.fail = function (err, code) {
204     var msg
205       , errObj;
206     if (code) {
207       jake.errorCode = code;
208     }
209     if (err) {
210       if (typeof err == 'string') {
211         // Use the initial or only line of the error as the error-message
212         // If there was a multi-line error, use the rest as the stack
213         msg = err.split('/n');
214         errObj = new Error(msg.shift());
215         if (msg.length) {
216           errObj.stack = msg.join('\n');
217         }
218         throw errObj;
219       }
220       else if (err instanceof Error) {
221         throw err;
222       }
223       else {
224         throw new Error(err.toString());
225       }
226     }
227     else {
228       throw new Error();
229     }
230   };
231 
232 })();
233
