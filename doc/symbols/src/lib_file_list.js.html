 18 var fs = require('fs')
 19 , path = require('path')
 20 , minimatch // Lazy-required
 21 , glob = {};

 23 glob.globSync = function (pat) {
 24 	var dirname = path.dirname(pat)
 25     , files = fs.readdirSync(dirname)
 26     , filePat = path.basename(pat)
 27     , matches;
 28 	matches = minimatch.match(files, filePat, {'null': true});
 29   if (matches && matches.length) {
 30     for (var i = 0, ii = matches.length; i < ii; i++) {
 31       matches[i] = dirname + '/' + matches[i];
 32     }
 33   }
 34   return matches
 35 };

 37 // Constants
 38 // ---------------
 39 // List of all the builtin Array methods we want to override
 40 var ARRAY_METHODS = Object.getOwnPropertyNames(Array.prototype)
 41 // Array methods that return a copy instead of affecting the original
 42   , SPECIAL_RETURN = {
 43       'concat': true
 44     , 'slice': true
 45     , 'filter': true
 46     , 'map': true
 47     }
 48 // Default file-patterns we want to ignore
 49   , DEFAULT_IGNORE_PATTERNS = [
 50       /(^|[\/\\])CVS([\/\\]|$)/
 51     , /(^|[\/\\])\.svn([\/\\]|$)/
 52     , /(^|[\/\\])\.git([\/\\]|$)/
 53     , /\.bak$/
 54     , /~$/
 55     ]
 56 // Ignore core files
 57   , DEFAULT_IGNORE_FUNCS = [
 58       function (name) {
 59         var isDir = false
 60           , stats;
 61         try {
 62           stats = fs.statSync(name);
 63           isDir = stats.isDirectory();
 64         }
 65         catch(e) {}
 66         return (/(^|[\/\\])core$/).test(name) && !isDir;
 67       }
 68     ];

 70 // Utility methods
 71 // ---------------
 72 // Escapes special characters in a string to be used in generating
 73 // a regex
 74 var regexpEscape = (function() {
 75   var specials = [ '/', '.', '*', '+', '?', '|', '(', ')', '[', ']', '{', '}', '\\' ]
 76   , sRE = new RegExp('(\\' + specials.join('|\\') + ')', 'g');
 77   return function (text) { return text.replace(sRE, '\\$1'); };
 78 })();

 81 var FileList = function () {
 82   var self = this
 83     , wrap;
 84 
 85   // Lazy-require minimatch so that require of this file in cli.js won't bomb
 86   if (!minimatch) {
 87     try {
 88       minimatch = require('minimatch');
 89     }
 90     catch(e) {
 91       // Hacky fallback, assume global install at /usr/local?
 92       try {
 93         minimatch = require('/usr/local/lib/node_modules/minimatch');
 94       }
 95       catch(e) {
 96         fail('FileList requires minimatch ' +
 97             '(https://github.com/isaacs/minimatch). Try `npm install -g minimatch`.');
 98       }
 99     }
100   }
101 
102   // List of glob-patterns or specific filenames
103   this.pendingAdd = [];
104   // Switched to false after lazy-eval of files
105   this.pending = true;
106   // Used to calculate exclusions from the list of files
107   this.excludes = {
108     pats: DEFAULT_IGNORE_PATTERNS.slice()
109   , funcs: DEFAULT_IGNORE_FUNCS.slice()
110   , regex: null
111   };
112   this.items = [];
113 
114   // Wrap the array methods with the delegates
115   wrap = function (prop) {
116     var arr;
117     self[prop] = function () {
118       if (self.pending) {
119         self.resolve();
120       }
121       if (typeof self.items[prop] == 'function') {
122         // Special method that return a copy
123         if (SPECIAL_RETURN[prop]) {
124           arr = self.items[prop].apply(self.items, arguments);
125           return FileList.clone(self, arr);
126         }
127         else {
128           return self.items[prop].apply(self.items, arguments);
129         }
130       }
131       else {
132         return self.items[prop];
133       }
134     };
135   };
136   for (var i = 0, ii = ARRAY_METHODS.length; i < ii; i++) {
137     wrap(ARRAY_METHODS[i]);
138   }
139 
140   // Include whatever files got passed to the constructor
141   this.include.apply(this, arguments);
142 
143   // Fix constructor linkage
144   this.constructor = FileList;
145 };
146 
147 FileList.prototype = new (function () {
148   var globPattern = /[*?\[\{]/;
149 
150   var _addMatching = function (pat) {
151         var matches = glob.globSync(pat);
152         this.items = this.items.concat(matches);
153       }
154 
155     , _resolveAdd = function (name) {
156         if (globPattern.test(name)) {
157           _addMatching.call(this, name);
158         }
159         else {
160           this.push(name);
161         }
162       }
163 
164     , _calculateExcludeRe = function () {
165         var pats = this.excludes.pats
166           , pat
167           , excl = []
168           , matches = [];
169 
170         for (var i = 0, ii = pats.length; i < ii; i++) {
171           pat = pats[i];
172           if (typeof pat == 'string') {
173             // Glob, look up files
174             if (/[*?]/.test(pat)) {
175               matches = glob.globSync(pat);
176               excl = excl.concat(matches);
177             }
178             // String for regex
179             else {
180               excl.push(regexpEscape(pat));
181             }
182           }
183           // Regex, grab the string-representation
184           else if (pat instanceof RegExp) {
185 186             excl.push(pat.toString().replace(/^\/|\/$/g, ''));
187           }
188         }
189         if (excl.length) {
190           this.excludes.regex = new RegExp('(' + excl.join(')|(') + ')');
191         }
192         else {
193           this.excludes.regex = /^$/;
194         }
195       }
196 
197     , _resolveExclude = function () {
198         var self = this;
199         _calculateExcludeRe.call(this);
200         // No `reject` method, so use reverse-filter
201         this.items = this.items.filter(function (name) {
202           return !self.shouldExclude(name);
203         });
204       }
205 
206   /**
207    * Includes file-patterns in the FileList. Should be called with one or more
208    * pattern for finding file to include in the list. Arguments should be strings
209    * for either a glob-pattern or a specific file-name, or an array of them
210    */
211   this.include = function () {
212     var args = Array.isArray(arguments[0]) ? arguments[0] : arguments;
213     for (var i = 0, ii = args.length; i < ii; i++) {
214       this.pendingAdd.push(args[i]);
215     }
216   };
217 
218   /**
219    * Indicates whether a particular file would be filtered out by the current
220    * exclusion rules for this FileList.
221    * @param {String} name The filename to check
222    * @return {Boolean} Whether or not the file should be excluded
223    */
224   this.shouldExclude = function (name) {
225     if (!this.excludes.regex) {
226       _calculateExcludeRe.call(this);
227     }
228     var excl = this.excludes;
229     return excl.regex.test(name) || excl.funcs.some(function (f) {
230       return !!f(name);
231     });
232   };
233 
234   /**
235    * Excludes file-patterns from the FileList. Should be called with one or more
236    * pattern for finding file to include in the list. Arguments can be:
237    * 1. Strings for either a glob-pattern or a specific file-name
238    * 2. Regular expression literals
239    * 3. Functions to be run on the filename that return a true/false
240    */
241   this.exclude = function () {
242     var args = Array.isArray(arguments[0]) ? arguments[0] : arguments
243       , arg;
244     for (var i = 0, ii = args.length; i < ii; i++) {
245       arg = args[i];
246       if (typeof arg == 'function' && !(arg instanceof RegExp)) {
247         this.excludes.funcs.push(arg);
248       }
249       else {
250         this.excludes.pats.push(arg);
251       }
252     }
253     if (!this.pending) {
254       _resolveExclude.call(this);
255     }
256   };
257 
258   /**
259    * Populates the FileList from the include/exclude rules with a list of
260    * actual files
261    */
262   this.resolve = function () {
263     var name;
264     if (this.pending) {
265       this.pending = false;
266       while ((name = this.pendingAdd.shift())) {
267         _resolveAdd.call(this, name);
268       }
269       _resolveExclude.call(this);
270     }
271   };
272 
273   /**
274    * Convert to a plain-jane array
275    */
276   this.toArray = function () {
277     // Call slice to ensure lazy-resolution before slicing items
278     var ret = this.slice().items.slice();
279     return ret;
280   };
281 
282   /**
283    * Get rid of any current exclusion rules
284    */
285   this.clearExclude = function () {
286     this.excludes = {
287       pats: []
288     , funcs: []
289     , regex: null
290     };
291   };
292 
293 })();

295 // Static method, used to create copy returned by special
296 // array methods
297 FileList.clone = function (list, items) {
298   var clone = new FileList();
299   if (items) {
300     clone.items = items;
301   }
302   clone.pendingAdd = list.pendingAdd;
303   clone.pending = list.pending;
304   for (var p in list.excludes) {
305     clone.excludes[p] = list.excludes[p];
306   }
307   return clone;
308 };

310 exports.FileList = FileList;
